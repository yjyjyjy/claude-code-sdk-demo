for room_type, count in zero_by_room_type.items():

zero_by_neighborhood = zero_reviews.groupby('listing_neighborhood').size().sort_values(ascending=False)

for neighborhood, count in zero_by_neighborhood.head(8).items():
    if neighborhood != '-unknown-':
        total_in_neighborhood = (df['listing_neighborhood'] == neighborhood).sum()
        print(f"     * {neighborhood}: {count:,} out of {total_in_neighborhood:,} ({count/total_in_neighborhood*100:.1f}%)")

# Calculate business impact and sizing
print("=== BUSINESS IMPACT SIZING ===\n")

# Current market size and performance
total_listings = len(df)
total_reviews = df[df['total_reviews'] >= 0]['total_reviews'].sum()  # Exclude negative values
avg_reviews_per_active_listing = df[df['total_reviews'] > 0]['total_reviews'].mean()

print(f"CURRENT MARKET METRICS:")
print(f"- Total listings: {total_listings:,}")
print(f"- Active listings (>0 reviews): {(df['total_reviews'] > 0).sum():,}")
print(f"- Total reviews generated: {int(total_reviews):,}")
print(f"- Average reviews per active listing: {avg_reviews_per_active_listing:.1f}")

# Growth opportunity sizing
print(f"\nGROWTH OPPORTUNITY SIZING:")

# Opportunity 1: Activate zero-review listings
zero_review_listings = (df['total_reviews'] == 0).sum()
potential_reviews_from_activation = zero_review_listings * avg_reviews_per_active_listing * 0.3  # Assume 30% activation rate
print(f"1. LISTING ACTIVATION:")
print(f"   - Target: {zero_review_listings:,} zero-review listings")
print(f"   - If 30% become active at avg performance: +{int(potential_reviews_from_activation):,} reviews")
print(f"   - Review growth potential: +{(potential_reviews_from_activation/total_reviews)*100:.1f}%")

# Opportunity 2: Neighborhood classification
unknown_listings_count = (df['listing_neighborhood'] == '-unknown-').sum()
known_vs_unknown_performance = df[df['listing_neighborhood'] != '-unknown-']['total_reviews'].mean() / df[df['listing_neighborhood'] == '-unknown-']['total_reviews'].mean()
potential_reviews_from_classification = unknown_listings_count * df[df['listing_neighborhood'] == '-unknown-']['total_reviews'].mean() * (known_vs_unknown_performance - 1)
print(f"\n2. NEIGHBORHOOD CLASSIFICATION:")
print(f"   - Target: {unknown_listings_count:,} unclassified listings")
print(f"   - Performance gap: {known_vs_unknown_performance:.1f}x better for classified listings")
print(f"   - Potential additional reviews: +{int(potential_reviews_from_classification):,}")
print(f"   - Review growth potential: +{(potential_reviews_from_classification/total_reviews)*100:.1f}%")

# Opportunity 3: Market expansion in underrepresented high-engagement neighborhoods
high_engagement_small_markets = underrepresented_neighborhoods.head(5)
expansion_potential = high_engagement_small_markets['listing_count'].sum() * 2  # Double the listings
potential_reviews_from_expansion = expansion_potential * high_engagement_small_markets['avg_reviews'].mean()
print(f"\n3. MARKET EXPANSION (Top 5 Underrepresented Markets):")
print(f"   - Current listings in target neighborhoods: {int(high_engagement_small_markets['listing_count'].sum())}")
print(f"   - Potential to double supply: +{int(expansion_potential)} listings")
print(f"   - Expected reviews from expansion: +{int(potential_reviews_from_expansion):,}")
print(f"   - Review growth potential: +{(potential_reviews_from_expansion/total_reviews)*100:.1f}%")

# Total opportunity
total_opportunity = potential_reviews_from_activation + potential_reviews_from_classification + potential_reviews_from_expansion
print(f"\nTOTAL GROWTH OPPORTUNITY:")
print(f"- Combined potential: +{int(total_opportunity):,} additional reviews")
print(f"- Overall growth potential: +{(total_opportunity/total_reviews)*100:.1f}%")
print(f"- This could increase total platform activity by ~{(total_opportunity/total_reviews)*100:.0f}%")
